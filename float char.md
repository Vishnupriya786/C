# float char

```
#include<stdio.h>
int main(){
	
	int Number = 5;
	float DecimalNumber = 5.6;
	char Letter = 88;  // result will be ASCII (American Standard Code for Information Interchange)
	printf("%d\n",Number);
	printf("%f\n",DecimalNumber);
	printf("%c\n",Letter);
	
	return 0;
}
```

```
OUTPUT:
5
5.600000
X
```
